package config

func TestBaseURL_SchemeBehavior_B1Fixture(t *testing.T) {
	u := Upload{BaseURL_: "http://example.com/path"}
	if got := u.BaseURL(true); !strings.HasPrefix(got, "https://") {
		t.Fatalf("BaseURL(true) = %q, want an https:// prefix", got)
	}
	if got := u.BaseURL(false); strings.HasPrefix(got, "https://") || !strings.HasPrefix(got, "http://") {
		t.Fatalf("BaseURL(false) = %q, want an http:// (non-https) prefix", got)
	}
}
